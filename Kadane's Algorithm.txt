class Solution{

    // arr: input array
    // n: size of array
    //Function to find the sum of contiguous subarray with maximum sum.
    long maxSubarraySum(int arr[], int n){
        
        // Your code here
        int sum=0, max=-99999;
        for(int i=0;i<n;i++)
        {
            sum+=arr[i];
            if(max<sum)max=sum;
            if(sum<0)sum=0;
        }
        return max;
    }
    
}
